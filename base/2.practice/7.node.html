# 七.使用 NODE 项目

# 1.安装 node.js

docker pull node
1

# 2.修改后端项目

进入项目根目录创建Dockerfile文件

FROM node:8.0.0
MAINTAINER EOT
ADD . /app/
WORKDIR /app
RUN npm install
ENV HOST O.0.0.0
ENV PORT 8000
EXPOSE 8000
CMD ["npm","start"]

# 3.创建 image

在Dockerfile文件同目录中执行

docker build -t myproject .
1

# 4.启动镜像

docker run -d -p 8000:8000 myproject
1

此时可通过 IP:8000 访问服务器

# 5.查看运行的镜像

docker ps
