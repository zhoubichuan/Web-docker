# 七.使用 NODE 项目

# 1.制作镜像

# 1.1 安装 node.js

docker pull node:latest
1

# 1.2 修改后端项目

进入项目根目录创建Dockerfile文件

FROM node:latest
MAINTAINER EOT
ADD . /app/
WORKDIR /app
RUN npm install
ENV HOST O.0.0.0
ENV PORT 8000
EXPOSE 8000
CMD ["npm","start"]
